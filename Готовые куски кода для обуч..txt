#бесконечный цикл#
while True:
    #4пробела#
    question = input("Строка ввода:").lower() #условия должны быть с маленькой буквы и условия нечуствительны к регистру#
    #создание функции ввода и создание переменной question#
    if "привет" in question:         #маленькая буква и если подходит условие выводит приветствие#
        print("Привет!") #+4 пробела#
        #вывод приветствия#
    elif "пока" in question: #условие если подходит условие выводит пока и выходит из программы с помощью break#
        print("Пока!") #+4 пробела#
        break  #прерывает цикл#        #так же+4 пробела#
    else: #если не подходит ни одно условие то выводит это#
        print("Не понимаю")
